✅ Approve Customers via Firebase / Google Sheet

If admin login fails, you can approve customers directly from Firebase or Google Sheet:

F
Via Firebase: Go to console.firebase.google.com → skyflies-solar → Firestore → register collection → find the customer → change status from "Pending Approval" to "Approved"
G
Via Google Sheet: Open Skyflies Solar CRM sheet → Pending Registrations tab → find the customer row → change column I "Pending Approval" to "Approved"
+
Then manually add to Customer Logins: Go to Customer Logins tab → add a new row with: Customer ID (SKY-001), Name, Password (hashed), Phone, Email, City, and other details

function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var ss   = SpreadsheetApp.getActiveSpreadsheet();
    if (data.action === 'getCustomer')      return respond(getCustomer(ss, data));
    if (data.action === 'addCustomer')      return respond(addCustomer(ss, data));
    if (data.action === 'addQuery')         return respond(addQuery(ss, data));
    if (data.action === 'registerCustomer') return respond(saveRegistration(ss, data));
    if (data.action === 'loginEvent')       return respond(saveLoginEvent(ss, data));
    if (data.action === 'getPending')       return respond(getPending(ss, data));
    if (data.action === 'sendOTP')          return respond(sendOTP(ss, data));
    if (data.action === 'getCustomerByEmail') return respond(getCustomerByEmail(ss, data));
    if (data.action === 'updatePassword')   return respond(updatePassword(ss, data));
    if (data.action === 'approveCustomer')  return respond(approveCustomer(ss, data));
    if (data.action === 'getAllCustomers')   return respond(getAllCustomers(ss, data));
    if (data.action === 'sendWelcomeEmail')  return respond(sendWelcomeEmail(ss, data));
    if (data.action === 'adminAlert')        return respond(sendAdminAlert(data));
    return respond(handleForm(ss, data));
  } catch(err) {
    return respond({status:'error', message:err.toString()});
  }
}

  return {status:'success',sheet:cfg.name};
}
