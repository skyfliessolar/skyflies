📄 Google Apps Script Code

Copy and paste into Apps Script, then deploy as Web App:

function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var map = { enquiry:'Enquiries', quotation:'Quotations', subscribe:'Subscriptions' };
    var hdrs = {
      enquiry:   ['Timestamp','Name','Phone','Email','City','Property','Source','Message'],
      quotation: ['Timestamp','Name','Phone','Email','Company','City','Property','Bill','Roof Area','System','Budget','Services','Message'],
      subscribe: ['Timestamp','Name','Phone','Email','Company','City','Address','kW','Panels','Plan','Billing','Service Day','Message']
    };
    var sn = map[data.formType] || 'General';
    var sh = ss.getSheetByName(sn);
    if (!sh) {
      sh = ss.insertSheet(sn);
      sh.appendRow(hdrs[data.formType] || ['Timestamp','Data']);
      sh.getRange(1,1,1,sh.getLastColumn()).setBackground('#F5A623').setFontColor('#000').setFontWeight('bold');
      sh.setFrozenRows(1);
    }
    var ts = new Date().toLocaleString('en-IN',{timeZone:'Asia/Kolkata'});
    if (data.formType==='enquiry')
      sh.appendRow([ts,data.name,data.phone,data.email,data.city,data.propertyType,data.source,data.message]);
    else if (data.formType==='quotation')
      sh.appendRow([ts,data.name,data.phone,data.email,data.company,data.city,data.propertyType,data.bill,data.roofArea,data.systemType,data.budget,data.services,data.message]);
    else if (data.formType==='subscribe')
      sh.appendRow([ts,data.name,data.phone,data.email,data.company,data.city,data.address,data.systemSize,data.panels,data.plan,data.billing,data.serviceDay,data.message]);
    sh.autoResizeColumns(1, sh.getLastColumn());
    MailApp.sendEmail('info@skyflies.in',
      '🌞 New '+data.formType+' – Skyflies Solar | '+data.name+' – '+data.city,
      'New submission:\n\nName: '+data.name+'\nPhone: '+data.phone+'\nCity: '+data.city+'\nType: '+data.formType+(data.plan?'\nPlan: '+data.plan:'')+'\n\nOpen sheet: '+ss.getUrl());
    return ContentService.createTextOutput(JSON.stringify({status:'success'})).setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({status:'error',message:err.toString()})).setMimeType(ContentService.MimeType.JSON);
  }
}
function doGet() {
  return ContentService.createTextOutput(JSON.stringify({status:'ok',app:'Skyflies Solar CRM'})).setMimeType(ContentService.MimeType.JSON);
}
